	\begin{definition}[正规子群]
		设$G$是群，$H<G$，若有
		$$ghg^{-1}\in H,\ \forall\ g\in G,\ \forall\ h\in H,$$
		则称$H$是$G$的\textbf{正规子群}，记作$H\vartriangleleft G$.
	\end{definition}
	\begin{theorem}
		设$H<G$，则下列条件等价.
		\begin{enumerate}
			\item $H\vartriangleleft G$;
			\item $gH=Hg,\ \forall\ g\in G$;
			\item $g_1Hg_2H=g_1g_2H,\ \forall\ g_1,g_2\in G$.
		\end{enumerate}
	\end{theorem}
